# 缓冲区溢出分析报告

## 源代码


    #define _CRT_SECURE_NO_WARNINGS

    #include <stdlib.h>
    #include <stdio.h>
    #include <string.h>

    int sub(char* x)
    {
      char y[10];
      strcpy(y, x);
      return 0;
    }

    int main(int argc, char** argv)
    {
      if (argc > 1)
      sub(argv[1]);
      printf("exit");
    }
